4-пример
#include<iostream>
using namespace std;
float one(int number)
{
	int count = 0;
	while (number!=0)
	{
		int temp = number % 10;
		number /= 10;
		if (temp==1) count++;
	}
	return count;
}
float main()
{
	int a, b, c;
	cin >> a >> b >> c;
	int countA = one(a);
	int countB = one(b);
	int countC = one(c);
	if (countA > countB && countA > countC) cout << a; else
		if (countB > countC) cout << b; else cout << c;
	system("pause");
}
